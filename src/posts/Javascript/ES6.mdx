- code block syntax 적용 확인
```javascript
/**
 * @param {Object} person - 사람 
 * @param {string} person.name - 이름
 * @param {number} person.age - 나이
 * @returns - 사람 정보
 */
const getPersonalInfo = (person) => {
  return person.name + ", " + person.age
}
```
